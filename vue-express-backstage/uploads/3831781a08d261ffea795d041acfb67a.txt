
create function nextval(sname VARCHAR(50)) RETURNS integer(11) 
begin 
 update sequence set current_val = current_val+increment_val where seq_name = sname ;
return currval(sname) ;
ends;


 create function currval(sname VARCHAR(50)) RETURNS INTEGER(11)
BEGIN
DECLARE value integer ;
set value = 0 ;
select current_val into value from sequence where seq_name = sname;
 return value ;
 end;
